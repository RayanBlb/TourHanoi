string affichage_terrain(int taille_disque, int nb_disque)
{
  int nb_espaces;
  stringstream Stream;

  if (taille_disque == 0)
  {
    nb_espaces = nb_disque + 1;
    Stream << string(nb_espaces, ' ') << "|" << string(nb_espaces, ' ');
    return Stream.str();
  }
  else
  {
    nb_espaces = nb_disque - taille_disque;
    Stream << string(nb_espaces, ' ') << "[" << string(taille_disque, '=') << taille_disque << string(taille_disque, '=') << "]" << string(nb_espaces, ' ');
    return Stream.str();
  }
}
